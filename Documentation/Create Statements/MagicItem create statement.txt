CREATE TABLE MagicItem 
(
MIID INT(20) NOT NULL AUTO_INCREMENT,
name VARCHAR(255) NOT NULL, 
type VARCHAR(255) NOT NULL,  
weight VARCHAR(255), 
ac VARCHAR(255),
strength VARCHAR(255),
stealth VARCHAR(255),
rarity VARCHAR(255),
dmg1 VARCHAR(255),
dmg2 VARCHAR(255),
dmgType VARCHAR(255),
property VARCHAR(255),
dmgrange VARCHAR(255),
PRIMARY KEY (MIID)
);
